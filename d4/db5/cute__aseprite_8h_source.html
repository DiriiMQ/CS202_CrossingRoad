13 SUMMARY
14
15 cute_asesprite.h is a single-file header that implements some functions to
16 parse .ase/.aseprite files. The entire file is parsed all at once and some
17 structs are filled out then handed back to you.
18
19
20 LIMITATIONS
21
22 Only the "normal" blend mode for layers is supported. As a workaround try
23 using the "merge down" function in Aseprite to create a normal layer.
24 Supporting all blend modes would take too much code to be worth it.
25
26 Does not support very old versions of Aseprite (with old palette chunks
27 0x0004 or 0x0011). Also does not support deprecrated mask chunk.
28
29 sRGB and ICC profiles are parsed but completely ignored when blending
30 frames together. If you want these to be used when composing frames you
31 have to do this yourself.
32

52
53 Simply load an .ase or .aseprite file from disk or from memory like so.
54
55 ase_t* ase = cute_aseprite_load_from_file("data/player.aseprite", NULL);

59

66 anim.add_frame(frame->duration, frame->pixels);
67 }

69
70 Then free it up when done.
71
72 cute_aseprite_free(ase);

74
75 DATA STRUCTURES
76
77 Aseprite files have frames, layers, and cels. A single frame is one frame of an
78 animation, formed by blending all the cels of an animation together. There is
79 one cel per layer per frame. Each cel contains its own pixel data.
80
81 The frame's pixels are automatically assumed to have been blended by the `normal`
82 blend mode. A warning is emit if any other blend mode is encountered. Feel free
83 to update the pixels of each frame with your own implementation of blending
84 functions. The frame's pixels are merely provided like this for convenience.
85

90 shy about opening a GitHub issue if there's a crash! It's quite easy to update
91 the parser as long as you upload your .ase file that shows the bug.

458static uint64_t s_peak_bits(deflate_t* s, int num_bits_to_read)
459{
460 if (s->count < num_bits_to_read)
461 {
462 if (s->word_index < s->word_count)
463 {
464 uint32_t word = s->words[s->word_index++];
465 s->bits |= (uint64_t)word << s->count;
466 s->count += 32;
467 CUTE_ASEPRITE_ASSERT(s->word_index <= s->word_count);
468 }
469
470 else if (s->final_word_available)
471 {
472 uint32_t word = s->final_word;
473 s->bits |= (uint64_t)word << s->count;
474 s->count += s->bits_left;
475 s->final_word_available = 0;
476 }
477 }
478
479 return s->bits;
480}

482static uint32_t s_consume_bits(deflate_t* s, int num_bits_to_read)
483{
484 CUTE_ASEPRITE_ASSERT(s->count >= num_bits_to_read);
485 uint32_t bits = (uint32_t)(s->bits & (((uint64_t)1 << num_bits_to_read) - 1));
486 s->bits >>= num_bits_to_read;
487 s->count -= num_bits_to_read;
488 s->bits_left -= num_bits_to_read;
489 return bits;
490}

492static uint32_t s_read_bits(deflate_t* s, int num_bits_to_read)
493{
494 CUTE_ASEPRITE_ASSERT(num_bits_to_read <= 32);
495 CUTE_ASEPRITE_ASSERT(num_bits_to_read >= 0);
496 CUTE_ASEPRITE_ASSERT(s->bits_left > 0);
497 CUTE_ASEPRITE_ASSERT(s->count <= 64);
498 CUTE_ASEPRITE_ASSERT(!s_would_overflow(s, num_bits_to_read));
499 s_peak_bits(s, num_bits_to_read);
500 uint32_t bits = s_consume_bits(s, num_bits_to_read);
501 return bits;
502}

504static uint32_t s_rev16(uint32_t a)
505{
506 a = ((a & 0xAAAA) >> 1) | ((a & 0x5555) << 1);
507 a = ((a & 0xCCCC) >> 2) | ((a & 0x3333) << 2);
508 a = ((a & 0xF0F0) >> 4) | ((a & 0x0F0F) << 4);
509 a = ((a & 0xFF00) >> 8) | ((a & 0x00FF) << 8);
510 return a;
511}

513// RFC 1951 section 3.2.2
514static uint32_t s_build(deflate_t* s, uint32_t* tree, uint8_t* lens, int sym_count)
515{
516 int n, codes[16], first[16], counts[16] = { 0 };
517 CUTE_ASEPRITE_UNUSED(s);
518
519 // Frequency count
520 for (n = 0; n < sym_count; n++) counts[lens[n]]++;
521
522 // Distribute codes
523 counts[0] = codes[0] = first[0] = 0;
524 for (n = 1; n <= 15; ++n)
525 {
526 codes[n] = (codes[n - 1] + counts[n - 1]) << 1;
527 first[n] = first[n - 1] + counts[n - 1];
528 }
529
530 for (uint32_t i = 0; i < (uint32_t)sym_count; ++i)
531 {
532 uint8_t len = lens[i];
533
534 if (len != 0)
535 {
536 CUTE_ASEPRITE_ASSERT(len < 16);
537 uint32_t code = (uint32_t)codes[len]++;
538 uint32_t slot = (uint32_t)first[len]++;
539 tree[slot] = (code << (32 - (uint32_t)len)) | (i << 4) | len;
540 }
541 }
542
543 return (uint32_t)first[15];
544}

546static int s_stored(deflate_t* s)
547{
548 char* p;
549
550 // 3.2.3
551 // skip any remaining bits in current partially processed byte
552 s_read_bits(s, s->count & 7);
553
554 // 3.2.4
555 // read LEN and NLEN, should complement each other
556 uint16_t LEN = (uint16_t)s_read_bits(s, 16);
557 uint16_t NLEN = (uint16_t)s_read_bits(s, 16);
558 uint16_t TILDE_NLEN = ~NLEN;
559 CUTE_ASEPRITE_CHECK(LEN == TILDE_NLEN, "Failed to find LEN and NLEN as complements within stored (uncompressed) stream.");
560 CUTE_ASEPRITE_CHECK(s->bits_left / 8 <= (int)LEN, "Stored block extends beyond end of input stream.");
561 p = s_ptr(s);
562 CUTE_ASEPRITE_MEMCPY(s->out, p, LEN);
563 s->out += LEN;
564 return 1;
565
566ase_err:
567 return 0;
568}

578static int s_decode(deflate_t* s, uint32_t* tree, int hi)
579{
580 uint64_t bits = s_peak_bits(s, 16);
581 uint32_t search = (s_rev16((uint32_t)bits) << 16) | 0xFFFF;
582 int lo = 0;
583 while (lo < hi)
584 {
585 int guess = (lo + hi) >> 1;
586 if (search < tree[guess]) hi = guess;
587 else lo = guess + 1;
588 }
589
590 uint32_t key = tree[lo - 1];
591 uint32_t len = (32 - (key & 0xF));
592 CUTE_ASEPRITE_ASSERT((search >> len) == (key >> len));
593
594 s_consume_bits(s, key & 0xF);
595 return (key >> 4) & 0xFFF;
596}

599static int s_dynamic(deflate_t* s)
600{
601 uint8_t lenlens[19] = { 0 };
602
603 uint32_t nlit = 257 + s_read_bits(s, 5);
604 uint32_t ndst = 1 + s_read_bits(s, 5);
605 uint32_t nlen = 4 + s_read_bits(s, 4);
606
607 for (uint32_t i = 0 ; i < nlen; ++i)
608 lenlens[s_permutation_order[i]] = (uint8_t)s_read_bits(s, 3);
609
610 // Build the tree for decoding code lengths
611 s->nlen = s_build(0, s->len, lenlens, 19);
612 uint8_t lens[288 + 32];
613
614 for (uint32_t n = 0; n < nlit + ndst;)
615 {
616 int sym = s_decode(s, s->len, (int)s->nlen);
617 switch (sym)
618 {
619 case 16: for (uint32_t i = 3 + s_read_bits(s, 2); i; --i, ++n) lens[n] = lens[n - 1]; break;
620 case 17: for (uint32_t i = 3 + s_read_bits(s, 3); i; --i, ++n) lens[n] = 0; break;
621 case 18: for (uint32_t i = 11 + s_read_bits(s, 7); i; --i, ++n) lens[n] = 0; break;
622 default: lens[n++] = (uint8_t)sym; break;
623 }
624 }
625
626 s->nlit = s_build(s, s->lit, lens, (int)nlit);
627 s->ndst = s_build(0, s->dst, lens + nlit, (int)ndst);
628 return 1;
629}

631// 3.2.3
632static int s_block(deflate_t* s)
633{
634 while (1)
635 {
636 int symbol = s_decode(s, s->lit, (int)s->nlit);
637
638 if (symbol < 256)
639 {
640 CUTE_ASEPRITE_CHECK(s->out + 1 <= s->out_end, "Attempted to overwrite out buffer while outputting a symbol.");
641 *s->out = (char)symbol;
642 s->out += 1;
643 }
644
645 else if (symbol > 256)
646 {
647 symbol -= 257;
648 uint32_t length = s_read_bits(s, (int)(s_len_extra_bits[symbol])) + s_len_base[symbol];
649 int distance_symbol = s_decode(s, s->dst, (int)s->ndst);
650 uint32_t backwards_distance = s_read_bits(s, s_dist_extra_bits[distance_symbol]) + s_dist_base[distance_symbol];
651 CUTE_ASEPRITE_CHECK(s->out - backwards_distance >= s->begin, "Attempted to write before out buffer (invalid backwards distance).");
652 CUTE_ASEPRITE_CHECK(s->out + length <= s->out_end, "Attempted to overwrite out buffer while outputting a string.");
653 char* src = s->out - backwards_distance;
654 char* dst = s->out;
655 s->out += length;
656
657 switch (backwards_distance)
658 {
659 case 1: // very common in images
660 CUTE_ASEPRITE_MEMSET(dst, *src, (size_t)length);
661 break;
662 default: while (length--) *dst++ = *src++;
663 }
664 }
665
666 else break;
667 }
668
669 return 1;
670
671ase_err:
672 return 0;
673}

675// 3.2.3
676static int s_inflate(const void* in, int in_bytes, void* out, int out_bytes, void* mem_ctx)
677{
678 CUTE_ASEPRITE_UNUSED(mem_ctx);
679 deflate_t* s = (deflate_t*)CUTE_ASEPRITE_ALLOC(sizeof(deflate_t), mem_ctx);
680 s->bits = 0;
681 s->count = 0;
682 s->word_index = 0;
683 s->bits_left = in_bytes * 8;
684
685 // s->words is the in-pointer rounded up to a multiple of 4
686 int first_bytes = (int)((((size_t)in + 3) & (size_t)(~3)) - (size_t)in);
687 s->words = (uint32_t*)((char*)in + first_bytes);
688 s->word_count = (in_bytes - first_bytes) / 4;
689 int last_bytes = ((in_bytes - first_bytes) & 3);
690
691 for (int i = 0; i < first_bytes; ++i)
692 s->bits |= (uint64_t)(((uint8_t*)in)[i]) << (i * 8);
693
694 s->final_word_available = last_bytes ? 1 : 0;
695 s->final_word = 0;
696 for(int i = 0; i < last_bytes; i++)
697 s->final_word |= ((uint8_t*)in)[in_bytes - last_bytes + i] << (i * 8);
698
699 s->count = first_bytes * 8;
700
701 s->out = (char*)out;
702 s->out_end = s->out + out_bytes;
703 s->begin = (char*)out;
704
705 int count = 0;
706 uint32_t bfinal;
707 do
708 {
709 bfinal = s_read_bits(s, 1);
710 uint32_t btype = s_read_bits(s, 2);
711
712 switch (btype)
713 {
714 case 0: CUTE_ASEPRITE_CALL(s_stored(s)); break;
715 case 1: s_fixed(s); CUTE_ASEPRITE_CALL(s_block(s)); break;
716 case 2: s_dynamic(s); CUTE_ASEPRITE_CALL(s_block(s)); break;
717 case 3: CUTE_ASEPRITE_CHECK(0, "Detected unknown block type within input stream.");
718 }
719
720 ++count;
721 }
722 while (!bfinal);
723
724 CUTE_ASEPRITE_FREE(s, mem_ctx);
725 return 1;
726
727ase_err:
728 CUTE_ASEPRITE_FREE(s, mem_ctx);
729 return 0;
730}

867static int s_mul_un8(int a, int b)
868{
869 int t = (a * b) + 0x80;
870 return (((t >> 8) + t) >> 8);
871}

873static ase_color_t s_blend(ase_color_t src, ase_color_t dst, uint8_t opacity)
874{
875 src.a = (uint8_t)s_mul_un8(src.a, opacity);
876 int a = src.a + dst.a - s_mul_un8(src.a, dst.a);
877 int r, g, b;
878 if (a == 0) {
879 r = g = b = 0;
880 } else {
881 r = dst.r + (src.r - dst.r) * src.a / a;
882 g = dst.g + (src.g - dst.g) * src.a / a;
883 b = dst.b + (src.b - dst.b) * src.a / a;
884 }
885 ase_color_t ret = { (uint8_t)r, (uint8_t)g, (uint8_t)b, (uint8_t)a };
886 return ret;
887}

899static ase_color_t s_color(ase_t* ase, void* src, int index)
900{
901 ase_color_t result;
902 if (ase->mode == ASE_MODE_RGBA) {
903 result = ((ase_color_t*)src)[index];
904 } else if (ase->mode == ASE_MODE_GRAYSCALE) {
905 uint8_t saturation = ((uint8_t*)src)[index * 2];
906 uint8_t a = ((uint8_t*)src)[index * 2 + 1];
907 result.r = result.g = result.b = saturation;
908 result.a = a;
909 } else {
910 CUTE_ASEPRITE_ASSERT(ase->mode == ASE_MODE_INDEXED);
911 uint8_t palette_index = ((uint8_t*)src)[index];
912 if (palette_index == ase->transparent_palette_entry_index) {
913 result.r = 0;
914 result.g = 0;
915 result.b = 0;
916 result.a = 0;
917 } else {
918 result = ase->palette.entries[palette_index].color;
919 }
920 }
921 return result;
922}

924ase_t* cute_aseprite_load_from_memory(const void* memory, int size, void* mem_ctx)
925{
926 ase_t* ase = (ase_t*)CUTE_ASEPRITE_ALLOC(sizeof(ase_t), mem_ctx);
927 CUTE_ASEPRITE_MEMSET(ase, 0, sizeof(*ase));
928
929 ase_state_t state = { 0 };
930 ase_state_t* s = &state;
931 s->in = (uint8_t*)memory;
932 s->end = s->in + size;
933 s->mem_ctx = mem_ctx;
934
935 s_skip(s, sizeof(uint32_t)); // File size.
936 int magic = (int)s_read_uint16(s);
937 CUTE_ASEPRITE_ASSERT(magic == 0xA5E0);
938
939 ase->frame_count = (int)s_read_uint16(s);
940 ase->w = s_read_uint16(s);
941 ase->h = s_read_uint16(s);
942 uint16_t bpp = s_read_uint16(s) / 8;
943 if (bpp == 4) ase->mode = ASE_MODE_RGBA;
944 else if (bpp == 2) ase->mode = ASE_MODE_GRAYSCALE;
945 else {
946 CUTE_ASEPRITE_ASSERT(bpp == 1);
947 ase->mode = ASE_MODE_INDEXED;
948 }
949 uint32_t valid_layer_opacity = s_read_uint32(s) & 1;
950 int speed = s_read_uint16(s);
951 s_skip(s, sizeof(uint32_t) * 2); // Spec says skip these bytes, as they're zero'd.
952 ase->transparent_palette_entry_index = s_read_uint8(s);
953 s_skip(s, 3); // Spec says skip these bytes.
954 ase->number_of_colors = (int)s_read_uint16(s);
955 ase->pixel_w = (int)s_read_uint8(s);
956 ase->pixel_h = (int)s_read_uint8(s);
957 ase->grid_x = (int)s_read_int16(s);
958 ase->grid_y = (int)s_read_int16(s);
959 ase->grid_w = (int)s_read_uint16(s);
960 ase->grid_h = (int)s_read_uint16(s);
961 s_skip(s, 84); // For future use (set to zero).
962
963 ase->frames = (ase_frame_t*)CUTE_ASEPRITE_ALLOC((int)(sizeof(ase_frame_t)) * ase->frame_count, mem_ctx);
964 CUTE_ASEPRITE_MEMSET(ase->frames, 0, sizeof(ase_frame_t) * (size_t)ase->frame_count);
965
966 ase_udata_t* last_udata = NULL;
967 int was_on_tags = 0;
968 int tag_index = 0;
969
970 ase_layer_t* layer_stack[CUTE_ASEPRITE_MAX_LAYERS];
971
972 // Parse all chunks in the .aseprite file.
973 for (int i = 0; i < ase->frame_count; ++i) {
974 ase_frame_t* frame = ase->frames + i;
975 frame->ase = ase;
976 s_skip(s, sizeof(uint32_t)); // Frame size.
977 magic = (int)s_read_uint16(s);
978 CUTE_ASEPRITE_ASSERT(magic == 0xF1FA);
979 int chunk_count = (int)s_read_uint16(s);
980 frame->duration_milliseconds = s_read_uint16(s);
981 if (frame->duration_milliseconds == 0) frame->duration_milliseconds = speed;
982 s_skip(s, 2); // For future use (set to zero).
983 uint32_t new_chunk_count = s_read_uint32(s);
984 if (new_chunk_count) chunk_count = (int)new_chunk_count;
985
986 for (int j = 0; j < chunk_count; ++j) {
987 uint32_t chunk_size = s_read_uint32(s);
988 uint16_t chunk_type = s_read_uint16(s);
989 chunk_size -= (uint32_t)(sizeof(uint32_t) + sizeof(uint16_t));
990 uint8_t* chunk_start = s->in;
991
992 switch (chunk_type) {
993 case 0x2004: // Layer chunk.
994 {
995 CUTE_ASEPRITE_ASSERT(ase->layer_count < CUTE_ASEPRITE_MAX_LAYERS);
996 ase_layer_t* layer = ase->layers + ase->layer_count++;
997 layer->flags = (ase_layer_flags_t)s_read_uint16(s);
998 layer->type = (ase_layer_type_t)s_read_uint16(s);
999 layer->parent = NULL;
1000 int child_level = (int)s_read_uint16(s);
1001 layer_stack[child_level] = layer;
1002 if (child_level) {
1003 layer->parent = layer_stack[child_level - 1];
1004 }
1005 s_skip(s, sizeof(uint16_t)); // Default layer width in pixels (ignored).
1006 s_skip(s, sizeof(uint16_t)); // Default layer height in pixels (ignored).
1007 int blend_mode = (int)s_read_uint16(s);
1008 if (blend_mode) CUTE_ASEPRITE_WARNING("Unknown blend mode encountered.");
1009 layer->opacity = s_read_uint8(s) / 255.0f;
1010 if (!valid_layer_opacity) layer->opacity = 1.0f;
1011 s_skip(s, 3); // For future use (set to zero).
1012 layer->name = s_read_string(s);
1013 last_udata = &layer->udata;
1014 } break;
1015
1016 case 0x2005: // Cel chunk.
1017 {
1018 CUTE_ASEPRITE_ASSERT(frame->cel_count < CUTE_ASEPRITE_MAX_LAYERS);
1019 ase_cel_t* cel = frame->cels + frame->cel_count++;
1020 int layer_index = (int)s_read_uint16(s);
1021 cel->layer = ase->layers + layer_index;
1022 cel->x = s_read_int16(s);
1023 cel->y = s_read_int16(s);
1024 cel->opacity = s_read_uint8(s) / 255.0f;
1025 int cel_type = (int)s_read_uint16(s);
1026 s_skip(s, 7); // For future (set to zero).
1027 switch (cel_type) {
1028 case 0: // Raw cel.
1029 cel->w = s_read_uint16(s);
1030 cel->h = s_read_uint16(s);
1031 cel->pixels = CUTE_ASEPRITE_ALLOC(cel->w * cel->h * bpp, mem_ctx);
1032 CUTE_ASEPRITE_MEMCPY(cel->pixels, s->in, (size_t)(cel->w * cel->h * bpp));
1033 s_skip(s, cel->w * cel->h * bpp);
1034 break;
1035
1036 case 1: // Linked cel.
1037 cel->is_linked = 1;
1038 cel->linked_frame_index = s_read_uint16(s);
1039 break;
1040
1041 case 2: // Compressed image cel.
1042 {
1043 cel->w = s_read_uint16(s);
1044 cel->h = s_read_uint16(s);
1045 int zlib_byte0 = s_read_uint8(s);
1046 int zlib_byte1 = s_read_uint8(s);
1047 int deflate_bytes = (int)chunk_size - (int)(s->in - chunk_start);
1048 void* pixels = s->in;
1049 CUTE_ASEPRITE_ASSERT((zlib_byte0 & 0x0F) == 0x08); // Only zlib compression method (RFC 1950) is supported.
1050 CUTE_ASEPRITE_ASSERT((zlib_byte0 & 0xF0) <= 0x70); // Innapropriate window size detected.
1051 CUTE_ASEPRITE_ASSERT(!(zlib_byte1 & 0x20)); // Preset dictionary is present and not supported.
1052 int pixels_sz = cel->w * cel->h * bpp;
1053 void* pixels_decompressed = CUTE_ASEPRITE_ALLOC(pixels_sz, mem_ctx);
1054 int ret = s_inflate(pixels, deflate_bytes, pixels_decompressed, pixels_sz, mem_ctx);
1055 if (!ret) CUTE_ASEPRITE_WARNING(s_error_reason);
1056 cel->pixels = pixels_decompressed;
1057 s_skip(s, deflate_bytes);
1058 } break;
1059 }
1060 last_udata = &cel->udata;
1061 } break;
1062
1063 case 0x2006: // Cel extra chunk.
1064 {
1065 ase_cel_t* cel = frame->cels + frame->cel_count;
1066 cel->has_extra = 1;
1067 cel->extra.precise_bounds_are_set = (int)s_read_uint32(s);
1068 cel->extra.precise_x = s_read_fixed(s);
1069 cel->extra.precise_y = s_read_fixed(s);
1070 cel->extra.w = s_read_fixed(s);
1071 cel->extra.h = s_read_fixed(s);
1072 s_skip(s, 16); // For future use (set to zero).
1073 } break;
1074
1075 case 0x2007: // Color profile chunk.
1076 {
1077 ase->has_color_profile = 1;
1078 ase->color_profile.type = (ase_color_profile_type_t)s_read_uint16(s);
1079 ase->color_profile.use_fixed_gamma = (int)s_read_uint16(s) & 1;
1080 ase->color_profile.gamma = s_read_fixed(s);
1081 s_skip(s, 8); // For future use (set to zero).
1082 if (ase->color_profile.type == ASE_COLOR_PROFILE_TYPE_EMBEDDED_ICC) {
1083 // Use the embedded ICC profile.
1084 ase->color_profile.icc_profile_data_length = s_read_uint32(s);
1085 ase->color_profile.icc_profile_data = CUTE_ASEPRITE_ALLOC(ase->color_profile.icc_profile_data_length, mem_ctx);
1086 CUTE_ASEPRITE_MEMCPY(ase->color_profile.icc_profile_data, s->in, ase->color_profile.icc_profile_data_length);
1087 s->in += ase->color_profile.icc_profile_data_length;
1088 }
1089 } break;
1090
1091 case 0x2018: // Tags chunk.
1092 {
1093 ase->tag_count = (int)s_read_uint16(s);
1094 s_skip(s, 8); // For future (set to zero).
1095 CUTE_ASEPRITE_ASSERT(ase->tag_count < CUTE_ASEPRITE_MAX_TAGS);
1096 for (int k = 0; k < ase->tag_count; ++k) {
1097 ase_tag_t tag;
1098 tag.from_frame = (int)s_read_uint16(s);
1099 tag.to_frame = (int)s_read_uint16(s);
1100 tag.loop_animation_direction = (ase_animation_direction_t)s_read_uint8(s);
1101 s_skip(s, 8); // For future (set to zero).
1102 tag.r = s_read_uint8(s);
1103 tag.g = s_read_uint8(s);
1104 tag.b = s_read_uint8(s);
1105 s_skip(s, 1); // Extra byte (zero).
1106 tag.name = s_read_string(s);
1107 ase->tags[k] = tag;
1108 }
1109 was_on_tags = 1;
1110 } break;
1111
1112 case 0x2019: // Palette chunk.
1113 {
1114 ase->palette.entry_count = (int)s_read_uint32(s);
1115 CUTE_ASEPRITE_ASSERT(ase->palette.entry_count <= CUTE_ASEPRITE_MAX_PALETTE_ENTRIES);
1116 int first_index = (int)s_read_uint32(s);
1117 int last_index = (int)s_read_uint32(s);
1118 s_skip(s, 8); // For future (set to zero).
1119 for (int k = first_index; k <= last_index; ++k) {
1120 int has_name = s_read_uint16(s);
1121 ase_palette_entry_t entry;
1122 entry.color.r = s_read_uint8(s);
1123 entry.color.g = s_read_uint8(s);
1124 entry.color.b = s_read_uint8(s);
1125 entry.color.a = s_read_uint8(s);
1126 if (has_name) {
1127 entry.color_name = s_read_string(s);
1128 } else {
1129 entry.color_name = NULL;
1130 }
1131 CUTE_ASEPRITE_ASSERT(k < CUTE_ASEPRITE_MAX_PALETTE_ENTRIES);
1132 ase->palette.entries[k] = entry;
1133 }
1134 } break;
1135
1136 case 0x2020: // Udata chunk.
1137 {
1138 CUTE_ASEPRITE_ASSERT(last_udata || was_on_tags);
1139 if (was_on_tags && !last_udata) {
1140 CUTE_ASEPRITE_ASSERT(tag_index < ase->tag_count);
1141 last_udata = &ase->tags[tag_index++].udata;
1142 }
1143 int flags = (int)s_read_uint32(s);
1144 if (flags & 1) {
1145 last_udata->has_text = 1;
1146 last_udata->text = s_read_string(s);
1147 }
1148 if (flags & 2) {
1149 last_udata->color.r = s_read_uint8(s);
1150 last_udata->color.g = s_read_uint8(s);
1151 last_udata->color.b = s_read_uint8(s);
1152 last_udata->color.a = s_read_uint8(s);
1153 }
1154 last_udata = NULL;
1155 } break;
1156
1157 case 0x2022: // Slice chunk.
1158 {
1159 int slice_count = (int)s_read_uint32(s);
1160 int flags = (int)s_read_uint32(s);
1161 s_skip(s, sizeof(uint32_t)); // Reserved.
1162 const char* name = s_read_string(s);

1222 }

1225 }

1229 int cy = cel->y;
1230 int cw = cel->w;
1231 int ch = cel->h;

1233 int ct = -s_min(cy, 0);

1245 dst[dst_index] = result;
1246 }
1247 }
1248 }
1249 }

1251 ase->mem_ctx = mem_ctx;

1253}
