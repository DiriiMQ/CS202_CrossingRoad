227Aseprite LoadAsepriteFromMemory(unsigned char* fileData, unsigned int size) {
228 struct Aseprite aseprite;
229 aseprite.ase = 0;
230
231 if (!IsWindowReady()) {
232 TraceLog(LOG_ERROR, "ASEPRITE: Loading an Aseprite requires the Window to be running");
233 return aseprite;
234 }
235
236 ase_t* ase = cute_aseprite_load_from_memory(fileData, (int)size, 0);
237 if (ase == 0 || ase->frame_count == 0 || ase->w == 0 || ase->h == 0) {
238 TraceLog(LOG_ERROR, "ASEPRITE: Failed to load Aseprite");
239 return aseprite;
240 }
241
242 // Build the sprite's final image.
243 Image image = GenImageColor(ase->w * ase->frame_count, ase->h, BLANK);
244
245 // Load all frames.
246 for (int i = 0; i < ase->frame_count; i++) {
247 ase_frame_t* frame = ase->frames + i;
248 Rectangle dest = {(float)(i * ase->w), 0, (float)ase->w, (float)ase->h};
249 Rectangle src = {0, 0, (float)ase->w, (float)ase->h};
250 Image frameImage = {
251 .data = frame->pixels,
252 .width = ase->w,
253 .height = ase->h,
254 .mipmaps = 1,
255 .format = PIXELFORMAT_UNCOMPRESSED_R8G8B8A8
256 };
257 ImageDraw(&image, frameImage, src, dest, WHITE);
258 }
259
260 // Apply the transparent pixel.
261 int transparency = ase->transparent_palette_entry_index;
262 if (transparency >= 0 && transparency < ase->palette.entry_count) {
263 ase_color_t transparentColor = ase->palette.entries[transparency].color;
264 Color source = {
265 .r = transparentColor.r,
266 .g = transparentColor.g,
267 .b = transparentColor.b,
268 .a = transparentColor.a
269 };
270 ImageColorReplace(&image, source, BLANK);
271 }
272
273 // Create the Texture
274 Texture2D texture = LoadTextureFromImage(image);
275
276 // Now that the texture is loaded into the GPU, we can clear the image.
277 UnloadImage(image);
278
279 // Save the texture as the Aseprite context.
280 ase->mem_ctx = MemAlloc(sizeof(Texture2D));
281 Texture2D* texturePointer = (Texture2D*)ase->mem_ctx;
282 texturePointer->format = texture.format;
283 texturePointer->id = texture.id;
284 texturePointer->mipmaps = texture.mipmaps;
285 texturePointer->width = texture.width;
286 texturePointer->height = texture.height;
287 aseprite.ase = ase;
288 TraceLog(LOG_INFO, "ASEPRITE: Loaded successfully (%ix%i - %i frames)", ase->w, ase->h, ase->frame_count);
289
290 return aseprite;
291}

303Aseprite LoadAseprite(const char* fileName) {
304 unsigned int bytesRead;
305 unsigned char* fileData = LoadFileData(fileName, &bytesRead);
306 if (bytesRead == 0 || fileData == 0) {
307 TraceLog(LOG_ERROR, "ASEPRITE: Failed to load aseprite file \"%s\"", fileName);
308 struct Aseprite aseprite;
309 aseprite.ase = 0;
310 return aseprite;
311 }
312
313 Aseprite ase = LoadAsepriteFromMemory(fileData, bytesRead);
314 UnloadFileData(fileData);
315 return ase;

407void UnloadAseprite(Aseprite aseprite) {
408 ase_t* ase = aseprite.ase;
409 if (ase == 0) {
410 return;
411 }
412
413 // Destroy the texture.
414 if (ase->mem_ctx != 0) {
415 // Unload the texture from the GPU.
416 UnloadTexture(GetAsepriteTexture(aseprite));
417
418 // Unload the texture object.
419 MemFree(ase->mem_ctx);
420 }
421
422 // Destory the aseprite data.
423 cute_aseprite_free(ase);
424
425 TraceLog(LOG_INFO, "ASEPRITE: Unloaded Aseprite data successfully");
426}

505void UpdateAsepriteTag(AsepriteTag* tag) {
506 if (tag == 0 || tag->tag == 0 || tag->aseprite.ase == 0) {
507 TraceLog(LOG_WARNING, "ASEPRITE: Cannot update empty tag");
508 return;
509 }
510
511 // Don't progress if the tag is paused.
512 if (tag->paused) {
513 return;
514 }
515
516 ase_t* ase = tag->aseprite.ase;
517 ase_tag_t* aseTag = tag->tag;
518
519 // Count down the timer and see if it's time to update the frame.
520 tag->timer -= GetFrameTime() * tag->speed;
521 if (tag->timer > 0) {
522 return;
523 }
524
525 // Advance the frame and see if it's time to reset the position.
526 tag->currentFrame += tag->direction;
527 switch (aseTag->loop_animation_direction) {
528 case ASE_ANIMATION_DIRECTION_FORWARDS:
529 if (tag->currentFrame > aseTag->to_frame) {
530 if (tag->loop) {
531 tag->currentFrame = aseTag->from_frame;
532 } else {
533 tag->currentFrame = aseTag->to_frame;
534 tag->paused = true;
535 }
536 }
537 break;
538 case ASE_ANIMATION_DIRECTION_BACKWORDS:
539 if (tag->currentFrame < aseTag->from_frame) {
540 if (tag->loop) {
541 tag->currentFrame = aseTag->to_frame;
542 } else {
543 tag->currentFrame = aseTag->from_frame;
544 tag->paused = true;
545 }
546 }
547 break;
548 case ASE_ANIMATION_DIRECTION_PINGPONG:
549 if (tag->direction > 0) {
550 if (tag->currentFrame > aseTag->to_frame) {
551 tag->direction = -1;
552 if (tag->loop) {
553 tag->currentFrame = aseTag->to_frame - 1;
554 } else {
555 tag->currentFrame = aseTag->to_frame;
556 tag->paused = true;
557 }
558 }
559 } else {
560 if (tag->currentFrame < aseTag->from_frame) {
561 tag->direction = 1;
562 if (tag->loop) {
563 tag->currentFrame = aseTag->from_frame + 1;
564 } else {
565 tag->currentFrame = aseTag->from_frame;
566 tag->paused = true;
567 }
568 }
569 }
570 break;
571 }
572
573 // Reset the timer.
574 // TODO(RobLoach): Add the original tag->timer to make up the different in frame time?
575 tag->timer = (float)(ase->frames[tag->currentFrame].duration_milliseconds) / 1000.0f /* + tag->timer; */;
576}

654AsepriteTag LoadAsepriteTagFromIndex(Aseprite aseprite, int index) {
655 AsepriteTag tag = GenAsepriteTagDefault();
656
657 // Ensure the Aseprite exists.
658 ase_t* ase = aseprite.ase;
659 if (ase == 0) {
660 TraceLog(LOG_ERROR, "ASEPRITE: Asprite not loaded when attempting to load tag #%i", index);
661 return tag;
662 }
663
664 // Ensure the tag exists
665 if (index < 0 || index >= ase->tag_count) {
666 TraceLog(LOG_ERROR, "ASEPRITE: Tag index %i out of range for %i tags", index, ase->tag_count);
667 return tag;
668 }
669
670 // Base tag information
671 tag.aseprite.ase = aseprite.ase;
672 tag.tag = &ase->tags[index];
673
674 // Set up the frame range
675 tag.direction = 1;
676 tag.currentFrame = tag.tag->from_frame;
677 if (tag.tag->loop_animation_direction == ASE_ANIMATION_DIRECTION_BACKWORDS) {
678 tag.currentFrame = tag.tag->to_frame;
679 tag.direction = -1;
680 }
681
682 // Pause the animation if it's a one-frame tag
683 if (tag.tag->from_frame == tag.tag->to_frame) {
684 tag.paused = true;
685 }
686
687 // Timer in seconds
688 tag.timer = (float)(ase->frames[tag.currentFrame].duration_milliseconds) / 1000.0f;
689
690 // Color
691 tag.color.r = (unsigned char)tag.tag->r;
692 tag.color.g = (unsigned char)tag.tag->g;
693 tag.color.b = (unsigned char)tag.tag->b;
694
695 // Name
696 tag.name = (char*)tag.tag->name;
697
698 // Display a trace log about the aseprite tag
699 TraceLog(LOG_TRACE, "ASEPRITE: [ID %i] Asprite tag loaded successfully (%s)", index, tag.name);
700
701 return tag;
702}

715AsepriteTag LoadAsepriteTag(Aseprite aseprite, const char* name) {
716 AsepriteTag tag = GenAsepriteTagDefault();
717 ase_t* ase = aseprite.ase;
718 if (ase == 0) {
719 TraceLog(LOG_ERROR, "ASEPRITE: Cannot load tag \"%s\" from an Aseprite that's not loaded", name);
720 return tag;
721 }
722
723 // Loop through all tags to find the correct name.
724 for (int i = 0; i < ase->tag_count; i++) {
725 if (TextIsEqual(name, ase->tags[i].name)) {
726 return LoadAsepriteTagFromIndex(aseprite, i);
727 }
728 }
729
730 // Display a warning about the missing aseprite
731 TraceLog(LOG_WARNING, "ASEPRITE: Could not find tag \"%s\"", name);
732
733 return tag;
734}

754AsepriteSlice LoadAsepriteSlice(Aseprite aseprite, const char* name) {
