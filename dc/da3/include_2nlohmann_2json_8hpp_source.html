560 void destroy(value_t t)
561 {
562 if (t == value_t::array || t == value_t::object)
563 {
564 // flatten the current json_value to a heap-allocated stack
565 std::vector<basic_json> stack;
566
567 // move the top-level items to stack
568 if (t == value_t::array)
569 {
570 stack.reserve(array->size());
571 std::move(array->begin(), array->end(), std::back_inserter(stack));
572 }
573 else
574 {
575 stack.reserve(object->size());
576 for (auto&& it : *object)
577 {
578 stack.push_back(std::move(it.second));
579 }
580 }
581
582 while (!stack.empty())
583 {
584 // move the last item to local variable to be processed
585 basic_json current_item(std::move(stack.back()));
586 stack.pop_back();
587
588 // if current_item is array/object, move
589 // its children to the stack to be processed later
590 if (current_item.is_array())
591 {
592 std::move(current_item.m_value.array->begin(), current_item.m_value.array->end(), std::back_inserter(stack));
593
594 current_item.m_value.array->clear();
595 }
596 else if (current_item.is_object())
597 {
598 for (auto&& it : *current_item.m_value.object)
599 {
600 stack.push_back(std::move(it.second));
601 }
602
603 current_item.m_value.object->clear();
604 }
605
606 // it's now safe that current_item get destructed
607 // since it doesn't have any children
608 }
609 }
610
611 switch (t)
612 {
613 case value_t::object:
614 {
615 AllocatorType<object_t> alloc;
616 std::allocator_traits<decltype(alloc)>::destroy(alloc, object);
617 std::allocator_traits<decltype(alloc)>::deallocate(alloc, object, 1);
618 break;
619 }
620
621 case value_t::array:
622 {
623 AllocatorType<array_t> alloc;
624 std::allocator_traits<decltype(alloc)>::destroy(alloc, array);
625 std::allocator_traits<decltype(alloc)>::deallocate(alloc, array, 1);
626 break;
627 }
628
629 case value_t::string:
630 {
631 AllocatorType<string_t> alloc;
632 std::allocator_traits<decltype(alloc)>::destroy(alloc, string);
633 std::allocator_traits<decltype(alloc)>::deallocate(alloc, string, 1);
634 break;
635 }
636
637 case value_t::binary:
638 {
639 AllocatorType<binary_t> alloc;
640 std::allocator_traits<decltype(alloc)>::destroy(alloc, binary);
641 std::allocator_traits<decltype(alloc)>::deallocate(alloc, binary, 1);
642 break;
643 }
644
645 case value_t::null:
646 case value_t::boolean:
647 case value_t::number_integer:
648 case value_t::number_unsigned:
649 case value_t::number_float:
650 case value_t::discarded:
651 default:
652 {
653 break;
654 }
655 }

900 {
901 // check if each element is an array with two elements whose first
902 // element is a string
903 bool is_an_object = std::all_of(init.begin(), init.end(),
904 [](const detail::json_ref<basic_json>& element_ref)
905 {
906 return element_ref->is_array() && element_ref->size() == 2 && (*element_ref)[0].is_string();
907 });
908
909 // adjust type if type deduction is not wanted
910 if (!type_deduction)
911 {
912 // if array is wanted, do not create an object though possible
913 if (manual_type == value_t::array)
914 {
915 is_an_object = false;
916 }
917
918 // if object is wanted but impossible, throw an exception
919 if (JSON_HEDLEY_UNLIKELY(manual_type == value_t::object && !is_an_object))
920 {
921 JSON_THROW(type_error::create(301, "cannot create object from initializer list", basic_json()));
922 }
923 }
924
925 if (is_an_object)
926 {
927 // the initializer list is a list of pairs -> create object
928 m_type = value_t::object;
929 m_value = value_t::object;
930
931 for (auto& element_ref : init)
932 {
933 auto element = element_ref.moved_or_copied();
934 m_value.object->emplace(
935 std::move(*((*element.m_value.array)[0].m_value.string)),
936 std::move((*element.m_value.array)[1]));
937 }
938 }
939 else
940 {
941 // the initializer list describes an array -> create array
942 m_type = value_t::array;
943 m_value.array = create<array_t>(init.begin(), init.end());
944 }
945
946 set_parents();
947 assert_invariant();

4226 basic_json patch(const basic_json& json_patch) const
4227 {
4228 // make a working copy to apply the patch to
4229 basic_json result = *this;
4230
4231 // the valid JSON Patch operations
4232 enum class patch_operations {add, remove, replace, move, copy, test, invalid};
4233
4234 const auto get_op = [](const std::string & op)
4235 {
4236 if (op == "add")
4237 {
4238 return patch_operations::add;
4239 }
4240 if (op == "remove")
4241 {
4242 return patch_operations::remove;
4243 }
4244 if (op == "replace")
4245 {
4246 return patch_operations::replace;
4247 }
4248 if (op == "move")
4249 {
4250 return patch_operations::move;
4251 }
4252 if (op == "copy")
4253 {
4254 return patch_operations::copy;
4255 }
4256 if (op == "test")
4257 {
4258 return patch_operations::test;
4259 }
4260
4261 return patch_operations::invalid;
4262 };
4263
4264 // wrapper for "add" operation; add value at ptr
4265 const auto operation_add = [&result](json_pointer & ptr, basic_json val)
4266 {
4267 // adding to the root of the target document means replacing it
4268 if (ptr.empty())
4269 {
4270 result = val;
4271 return;
4272 }
4273
4274 // make sure the top element of the pointer exists
4275 json_pointer top_pointer = ptr.top();
4276 if (top_pointer != ptr)
4277 {
4278 result.at(top_pointer);
4279 }
4280
4281 // get reference to parent of JSON pointer ptr
4282 const auto last_path = ptr.back();
4283 ptr.pop_back();
4284 basic_json& parent = result[ptr];
4285
4286 switch (parent.m_type)
4287 {

4501 static basic_json diff(const basic_json& source, const basic_json& target,
4502 const std::string& path = "")
4503 {
4504 // the patch
4505 basic_json result(value_t::array);
4506
4507 // if the values are the same, return empty patch
4508 if (source == target)
4509 {
4510 return result;
4511 }
4512
4513 if (source.type() != target.type())
4514 {
4515 // different types: replace value
4516 result.push_back(
4517 {
4518 {"op", "replace"}, {"path", path}, {"value", target}
4519 });
4520 return result;
4521 }
4522
4523 switch (source.type())
4524 {
4525 case value_t::array:
4526 {
4527 // first pass: traverse common elements
4528 std::size_t i = 0;
4529 while (i < source.size() && i < target.size())
4530 {
4531 // recursive call to compare array values at index i
4532 auto temp_diff = diff(source[i], target[i], path + "/" + std::to_string(i));
4533 result.insert(result.end(), temp_diff.begin(), temp_diff.end());
4534 ++i;
4535 }
4536
4537 // We now reached the end of at least one array
4538 // in a second pass, traverse the remaining elements
4539
4540 // remove my remaining elements
4541 const auto end_index = static_cast<difference_type>(result.size());
4542 while (i < source.size())
4543 {
4544 // add operations in reverse order to avoid invalid
4545 // indices
4546 result.insert(result.begin() + end_index, object(
4547 {
4548 {"op", "remove"},
4549 {"path", path + "/" + std::to_string(i)}
4550 }));
4551 ++i;
4552 }
4553
4554 // add other remaining elements
4555 while (i < target.size())
4556 {
4557 result.push_back(
4558 {
4559 {"op", "add"},
4560 {"path", path + "/-"},
4561 {"value", target[i]}
4562 });
4563 ++i;
4564 }
4565
4566 break;
4567 }
4568
4569 case value_t::object:
4570 {
4571 // first pass: traverse this object's elements
4572 for (auto it = source.cbegin(); it != source.cend(); ++it)
4573 {
4574 // escape the key name to be used in a JSON patch
4575 const auto path_key = path + "/" + detail::escape(it.key());
4576
4577 if (target.find(it.key()) != target.end())
4578 {
4579 // recursive call to compare object values at key it
4580 auto temp_diff = diff(it.value(), target[it.key()], path_key);
4581 result.insert(result.end(), temp_diff.begin(), temp_diff.end());
4582 }
4583 else
4584 {
4585 // found a key that is not in o -> remove it
4586 result.push_back(object(
4587 {
4588 {"op", "remove"}, {"path", path_key}
4589 }));
4590 }
4591 }
4592
4593 // second pass: traverse other object's elements
4594 for (auto it = target.cbegin(); it != target.cend(); ++it)
4595 {
4596 if (source.find(it.key()) == source.end())
4597 {
4598 // found a key that is not in this -> add it
4599 const auto path_key = path + "/" + detail::escape(it.key());
4600 result.push_back(
4601 {
4602 {"op", "add"}, {"path", path_key},
4603 {"value", it.value()}
4604 });
4605 }
4606 }
4607
4608 break;
4609 }
4610
4611 case value_t::null:
4612 case value_t::string:
4613 case value_t::boolean:
4614 case value_t::number_integer:
4615 case value_t::number_unsigned:
4616 case value_t::number_float:
4617 case value_t::binary:
4618 case value_t::discarded:
4619 default:
4620 {
4621 // both primitive type: replace value
4622 result.push_back(
4623 {
4624 {"op", "replace"}, {"path", path}, {"value", target}
4625 });
4626 break;
4627 }
4628 }
4629
4630 return result;
4631 }
