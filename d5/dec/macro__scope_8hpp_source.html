159#define NLOHMANN_JSON_SERIALIZE_ENUM(ENUM_TYPE, ...) \
160 template<typename BasicJsonType> \
161 inline void to_json(BasicJsonType& j, const ENUM_TYPE& e) \
162 { \
163 static_assert(std::is_enum<ENUM_TYPE>::value, #ENUM_TYPE " must be an enum!"); \
164 static const std::pair<ENUM_TYPE, BasicJsonType> m[] = __VA_ARGS__; \
165 auto it = std::find_if(std::begin(m), std::end(m), \
166 [e](const std::pair<ENUM_TYPE, BasicJsonType>& ej_pair) -> bool \
167 { \
168 return ej_pair.first == e; \
169 }); \
170 j = ((it != std::end(m)) ? it : std::begin(m))->second; \
171 } \
172 template<typename BasicJsonType> \
173 inline void from_json(const BasicJsonType& j, ENUM_TYPE& e) \
174 { \
175 static_assert(std::is_enum<ENUM_TYPE>::value, #ENUM_TYPE " must be an enum!"); \
176 static const std::pair<ENUM_TYPE, BasicJsonType> m[] = __VA_ARGS__; \
177 auto it = std::find_if(std::begin(m), std::end(m), \
178 [&j](const std::pair<ENUM_TYPE, BasicJsonType>& ej_pair) -> bool \
179 { \
180 return ej_pair.second == j; \
181 }); \
182 e = ((it != std::end(m)) ? it : std::begin(m))->first; \
183 }
