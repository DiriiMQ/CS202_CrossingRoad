213// Returns the cameras forward vector (normalized)
214Vector3 GetCameraForward(Camera *camera)
215{
216 return Vector3Normalize(Vector3Subtract(camera->target, camera->position));
217}
218
219// Returns the cameras up vector (normalized)
220// Note: The up vector might not be perpendicular to the forward vector
221Vector3 GetCameraUp(Camera *camera)
222{
223 return Vector3Normalize(camera->up);
224}
225
226// Returns the cameras right vector (normalized)
227Vector3 GetCameraRight(Camera *camera)
228{
229 Vector3 forward = GetCameraForward(camera);
230 Vector3 up = GetCameraUp(camera);
231
232 return Vector3CrossProduct(forward, up);
233}

235// Moves the camera in its forward direction
236void CameraMoveForward(Camera *camera, float distance, bool moveInWorldPlane)
237{
238 Vector3 forward = GetCameraForward(camera);
239
240 if (moveInWorldPlane)
241 {
242 // Project vector onto world plane
243 forward.y = 0;
244 forward = Vector3Normalize(forward);
245 }
246
247 // Scale by distance
248 forward = Vector3Scale(forward, distance);
249
250 // Move position and target
251 camera->position = Vector3Add(camera->position, forward);
252 camera->target = Vector3Add(camera->target, forward);
253}
254
255// Moves the camera in its up direction
256void CameraMoveUp(Camera *camera, float distance)
257{
258 Vector3 up = GetCameraUp(camera);
259
260 // Scale by distance
261 up = Vector3Scale(up, distance);
262
263 // Move position and target
264 camera->position = Vector3Add(camera->position, up);
265 camera->target = Vector3Add(camera->target, up);
266}

268// Moves the camera target in its current right direction
269void CameraMoveRight(Camera *camera, float distance, bool moveInWorldPlane)
270{
271 Vector3 right = GetCameraRight(camera);
272
273 if (moveInWorldPlane)
274 {
275 // Project vector onto world plane
276 right.y = 0;
277 right = Vector3Normalize(right);
278 }
279
280 // Scale by distance
281 right = Vector3Scale(right, distance);
282
283 // Move position and target
284 camera->position = Vector3Add(camera->position, right);
285 camera->target = Vector3Add(camera->target, right);
286}

288// Moves the camera position closer/farther to/from the camera target
289void CameraMoveToTarget(Camera *camera, float delta)
290{
291 float distance = Vector3Distance(camera->position, camera->target);
292
293 // Apply delta
294 distance += delta;
295
296 // Distance must be greater than 0
297 if (distance < 0) distance = 0.001f;
298
299 // Set new distance by moving the position along the forward vector
300 Vector3 forward = GetCameraForward(camera);
301 camera->position = Vector3Add(camera->target, Vector3Scale(forward, -distance));
302}

304// Rotates the camera around its up vector
305// Yaw is "looking left and right"
306// If rotateAroundTarget is false, the camera rotates around its position
307// Note: angle must be provided in radians
308void CameraYaw(Camera *camera, float angle, bool rotateAroundTarget)
309{
310 // Rotation axis
311 Vector3 up = GetCameraUp(camera);
312
313 // View vector
314 Vector3 targetPosition = Vector3Subtract(camera->target, camera->position);
315
316 // Rotate view vector around up axis
317 targetPosition = Vector3RotateByAxisAngle(targetPosition, up, angle);
318
319 if (rotateAroundTarget)
320 {
321 // Move position relative to target
322 camera->position = Vector3Subtract(camera->target, targetPosition);
323 }
324 else // rotate around camera.position
325 {
326 // Move target relative to position
327 camera->target = Vector3Add(camera->position, targetPosition);
328 }
329}
330
331// Rotates the camera around its right vector, pitch is "looking up and down"
332// - lockView prevents camera overrotation (aka "somersaults")
333// - rotateAroundTarget defines if rotation is around target or around its position
334// - rotateUp rotates the up direction as well (typically only usefull in CAMERA_FREE)
335// NOTE: angle must be provided in radians
336void CameraPitch(Camera *camera, float angle, bool lockView, bool rotateAroundTarget, bool rotateUp)
337{
338 // Up direction
339 Vector3 up = GetCameraUp(camera);
340
341 // View vector
342 Vector3 targetPosition = Vector3Subtract(camera->target, camera->position);
343
344 if (lockView)
345 {
346 // In these camera modes we clamp the Pitch angle
347 // to allow only viewing straight up or down.
348
349 // Clamp view up
350 float maxAngleUp = Vector3Angle(up, targetPosition);
351 maxAngleUp -= 0.001f; // avoid numerical errors
352 if (angle > maxAngleUp) angle = maxAngleUp;
353
354 // Clamp view down
355 float maxAngleDown = Vector3Angle(Vector3Negate(up), targetPosition);
356 maxAngleDown *= -1.0f; // downwards angle is negative
357 maxAngleDown += 0.001f; // avoid numerical errors
358 if (angle < maxAngleDown) angle = maxAngleDown;
359 }
360
361 // Rotation axis
362 Vector3 right = GetCameraRight(camera);
363
364 // Rotate view vector around right axis
365 targetPosition = Vector3RotateByAxisAngle(targetPosition, right, angle);
366
367 if (rotateAroundTarget)
368 {
369 // Move position relative to target
370 camera->position = Vector3Subtract(camera->target, targetPosition);
371 }
372 else // rotate around camera.position
373 {
374 // Move target relative to position
375 camera->target = Vector3Add(camera->position, targetPosition);
376 }
377
378 if (rotateUp)
379 {
380 // Rotate up direction around right axis
381 camera->up = Vector3RotateByAxisAngle(camera->up, right, angle);
382 }
383}
384
385// Rotates the camera around its forward vector
386// Roll is "turning your head sideways to the left or right"
387// Note: angle must be provided in radians
388void CameraRoll(Camera *camera, float angle)
389{
390 // Rotation axis
391 Vector3 forward = GetCameraForward(camera);
392
393 // Rotate up direction around forward axis
394 camera->up = Vector3RotateByAxisAngle(camera->up, forward, angle);
395}

397// Returns the camera view matrix
398Matrix GetCameraViewMatrix(Camera *camera)
399{
400 return MatrixLookAt(camera->position, camera->target, camera->up);
401}
402
403// Returns the camera projection matrix
404Matrix GetCameraProjectionMatrix(Camera *camera, float aspect)
405{
406 if (camera->projection == CAMERA_PERSPECTIVE)
407 {
408 return MatrixPerspective(camera->fovy*DEG2RAD, aspect, CAMERA_CULL_DISTANCE_NEAR, CAMERA_CULL_DISTANCE_FAR);
409 }
410 else if (camera->projection == CAMERA_ORTHOGRAPHIC)
411 {
412 double top = camera->fovy/2.0;
413 double right = top*aspect;
414
415 return MatrixOrtho(-right, right, -top, top, CAMERA_CULL_DISTANCE_NEAR, CAMERA_CULL_DISTANCE_FAR);
416 }
417
418 return MatrixIdentity();
419}

422// Update camera position for selected mode
423// Camera mode: CAMERA_FREE, CAMERA_FIRST_PERSON, CAMERA_THIRD_PERSON, CAMERA_ORBITAL or CUSTOM
424void UpdateCamera(Camera *camera, int mode)
425{
426 Vector2 mousePositionDelta = GetMouseDelta();
427
428 bool moveInWorldPlane = ((mode == CAMERA_FIRST_PERSON) || (mode == CAMERA_THIRD_PERSON));
429 bool rotateAroundTarget = ((mode == CAMERA_THIRD_PERSON) || (mode == CAMERA_ORBITAL));
430 bool lockView = ((mode == CAMERA_FIRST_PERSON) || (mode == CAMERA_THIRD_PERSON) || (mode == CAMERA_ORBITAL));
431 bool rotateUp = (mode == CAMERA_FREE);
432
433 if (mode == CAMERA_ORBITAL)
434 {
435 // Orbital can just orbit
436 Matrix rotation = MatrixRotate(GetCameraUp(camera), CAMERA_ORBITAL_SPEED*GetFrameTime());
437 Vector3 view = Vector3Subtract(camera->position, camera->target);
438 view = Vector3Transform(view, rotation);
439 camera->position = Vector3Add(camera->target, view);
440 }
441 else
442 {
443 // Camera rotation
444 if (IsKeyDown(KEY_DOWN)) CameraPitch(camera, -CAMERA_ROTATION_SPEED, lockView, rotateAroundTarget, rotateUp);
445 if (IsKeyDown(KEY_UP)) CameraPitch(camera, CAMERA_ROTATION_SPEED, lockView, rotateAroundTarget, rotateUp);
446 if (IsKeyDown(KEY_RIGHT)) CameraYaw(camera, -CAMERA_ROTATION_SPEED, rotateAroundTarget);
447 if (IsKeyDown(KEY_LEFT)) CameraYaw(camera, CAMERA_ROTATION_SPEED, rotateAroundTarget);
448 if (IsKeyDown(KEY_Q)) CameraRoll(camera, -CAMERA_ROTATION_SPEED);
449 if (IsKeyDown(KEY_E)) CameraRoll(camera, CAMERA_ROTATION_SPEED);
450
451 CameraYaw(camera, -mousePositionDelta.x*CAMERA_MOUSE_MOVE_SENSITIVITY, rotateAroundTarget);
452 CameraPitch(camera, -mousePositionDelta.y*CAMERA_MOUSE_MOVE_SENSITIVITY, lockView, rotateAroundTarget, rotateUp);
453
454 // Camera movement
455 if (IsKeyDown(KEY_W)) CameraMoveForward(camera, CAMERA_MOVE_SPEED, moveInWorldPlane);
456 if (IsKeyDown(KEY_A)) CameraMoveRight(camera, -CAMERA_MOVE_SPEED, moveInWorldPlane);
457 if (IsKeyDown(KEY_S)) CameraMoveForward(camera, -CAMERA_MOVE_SPEED, moveInWorldPlane);
458 if (IsKeyDown(KEY_D)) CameraMoveRight(camera, CAMERA_MOVE_SPEED, moveInWorldPlane);
459 //if (IsKeyDown(KEY_SPACE)) CameraMoveUp(camera, CAMERA_MOVE_SPEED);
460 //if (IsKeyDown(KEY_LEFT_CONTROL)) CameraMoveUp(camera, -CAMERA_MOVE_SPEED);
461 }
462
463 if ((mode == CAMERA_THIRD_PERSON) || (mode == CAMERA_ORBITAL))
464 {
465 // Zoom target distance
466 CameraMoveToTarget(camera, -GetMouseWheelMove());
467 if (IsKeyPressed(KEY_KP_SUBTRACT)) CameraMoveToTarget(camera, 2.0f);
468 if (IsKeyPressed(KEY_KP_ADD)) CameraMoveToTarget(camera, -2.0f);
469 }
470}

473// Update camera movement, movement/rotation values should be provided by user
474void UpdateCameraPro(Camera *camera, Vector3 movement, Vector3 rotation, float zoom)
475{
476 // Required values
477 // movement.x - Move forward/backward
478 // movement.y - Move right/left
479 // movement.z - Move up/down
480 // rotation.x - yaw
481 // rotation.y - pitch
482 // rotation.z - roll
483 // zoom - Move towards target
484
485 bool lockView = true;
486 bool rotateAroundTarget = false;
487 bool rotateUp = false;
488 bool moveInWorldPlane = true;
489
490 // Camera rotation
491 CameraPitch(camera, -rotation.y*DEG2RAD, lockView, rotateAroundTarget, rotateUp);
492 CameraYaw(camera, -rotation.x*DEG2RAD, rotateAroundTarget);
493 CameraRoll(camera, rotation.z*DEG2RAD);
494
495 // Camera movement
496 CameraMoveForward(camera, movement.x, moveInWorldPlane);
497 CameraMoveRight(camera, movement.y, moveInWorldPlane);
498 CameraMoveUp(camera, movement.z);
499
500 // Zoom target distance
501 CameraMoveToTarget(camera, zoom);
502}
